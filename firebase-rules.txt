// Firebase Firestore Security Rules
// Add these to: Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // User books collection - users can only read/write their own books
    match /users/{userId}/books/{bookId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Universal books collection - publicly readable, only writable by admins
    match /books/{bookId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.role == 'admin';
    }

    // Users collection - users can only read/write their own profile
    match /users/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

// Firebase Storage Security Rules
// Add these to: Firebase Console -> Storage -> Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // User images - only the owner can read/write
    match /users/{userId}/images/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Book covers - publicly readable, only owner can write
    match /book-covers/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
